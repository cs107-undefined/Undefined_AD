    """[summary]

        ValueError: [description]
        ValueError: [description]

    try:
        temp = np.power(val, degree)
        if np.any(np.isnan(temp)):
            raise ValueError("invalid inputs for pow()")
    except ValueError as e:
        raise ValueError(e)
